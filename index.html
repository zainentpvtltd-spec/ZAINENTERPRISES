GST / FBR Tax

Sales tax tracking, input tax credit, and FBR return summary

🏛 Federal Board of Revenue
Pakistan Sales Tax Management · STRN: Not set
Standard rate: 17%
Pharma (OTC): 1%–17%
Exempt medicines: 0%
⚠ Pakistan Pharma Tax Note: Most prescription medicines are zero-rated (0% GST). OTC products may attract GST. Surgical supplies are generally taxable at 17%. Always verify with your FBR-registered accountant before filing. Your STRN is required to collect and remit GST.

Google Sheets setup

One-time 5-minute setup — all data syncs to your Google Sheet automatically

After setup, your data is stored in Google Sheets — accessible from any device and viewable as a spreadsheet anytime.

Step 1 — Create your Google Sheet

1Open Google Sheets and create a blank spreadsheet

Go to sheets.google.com → click Blank → name it: Zain Enterprises Distribution Data

Open Google Sheets ↗

Step 2 — Open Apps Script and paste the backend code

2In your sheet: Extensions → Apps Script → paste this code
function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(data.sheet) || ss.insertSheet(data.sheet); if (data.action === "write") { sheet.clearContents(); if (data.headers) sheet.appendRow(data.headers); data.rows.forEach(row => sheet.appendRow(row)); } else if (data.action === "append") { if (sheet.getLastRow() === 0 && data.headers) sheet.appendRow(data.headers); sheet.appendRow(data.row); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); }}function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(e.parameter.sheet); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ok:true,data:[]})) .setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({ok:true,data:rows})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); }}
